# Используем официальный образ SDK для сборки
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Копируем файлы проекта и восстанавливаем зависимости
COPY ["FitnessBot.csproj", "."]
RUN dotnet restore "FitnessBot.csproj"

# Копируем всё остальное и собираем приложение
COPY . .
RUN dotnet publish -c Release -o /app

# Используем официальный образ рантайма для запуска
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app .

# Ваше приложение слушает на порту, который укажет Render (обычно 10000)
ENV ASPNETCORE_URLS="http://*:$PORT"
# Сообщаем .NET, что мы находимся в Production-среде
ENV ASPNETCORE_ENVIRONMENT="Production"

# Запускаем приложение
ENTRYPOINT ["dotnet", "FitnessBot.dll"]
